64  // will automatically round down due to int division, subtract 1 due to 0 indexing
65  size_t mid = (v_size/2) - 1;
66  size_t midpo = mid + 1;
67  if (v_size % 2 == 1){
68  std::nth_element(vec.begin(), vec.begin()+(midpo), vec.end(),
69  [](T const a, T const b){

75  std::nth_element(vec.begin(), vec.begin()+(mid), vec.end(),

77  return b > a;
78  });

80  std::nth_element(vec.begin(), vec.begin()+(midpo), vec.end(),
81  [](T const a, T const b){
82  return b > a;
83  });

125 template<typename T>
126 std::vector<T> NumUtils::minMaxInVec(const std::vector<T> &vec, size_t begin, size_t end){
127  std::vector<T> M(2);
128  if(end == begin){
129  M[0] = vec[begin]; M[1] = vec[end];
130  return M;
131  }
132  if(end - begin == 1){
133  if(vec[begin] <= vec[end]){
134  M[0] = vec[begin]; M[1] = vec[end];
135  }
136  else{
137  M[0] = vec[end]; M[1] = vec[begin];
138  }
139  return M;
140  }
141  auto m = (end + begin) / 2;
142  std::vector<T> M1(2), M2(2);
143  M1 = minMaxInVec(vec, begin, m);
144  M2 = minMaxInVec(vec, m+1, end);
145  M[0] = std::min(M1[0], M2[0]);
146  M[1] = std::max(M1[1], M2[1]);
147  return M;
148 }
