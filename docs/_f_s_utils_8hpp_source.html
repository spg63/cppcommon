117 size_t FSUtils::lineCount(const std::string &fileName){
118  const char *fname = fileName.c_str();
119  static const auto BUF_SIZE = 16*1024;
120  int fd = open(fname, O_RDONLY);
121  if(fd == -1)
122  throw std::runtime_error("Couldn't open " + std::string(fname));
123 
124  char buf[BUF_SIZE + 1];
125  size_t lines = 0;
126 
127  while(size_t bytes_read = read(fd, buf, BUF_SIZE)){
128  if(bytes_read == (size_t) -1)
129  throw std::runtime_error("Read Failure" + std::string(fname));
130 
131  if(!bytes_read)
132  break;
133 
134  for(char *p = buf; (p = (char *) memchr(p, '\n', (buf + bytes_read) - p)); ++p)
135  ++lines;
136  }
137  close(fd);
138  return lines;
139 }

147 std::vector<std::string> FSUtils::getFilesInDir(const std::string &dirPath, const std::string &ext){
148  std::vector<std::string> files;
149 
150  DIR *dir;
151  struct dirent *ent;
152 
153  if((dir = opendir(dirPath.c_str())) == NULL)
154  throw std::runtime_error("Couldn't open " + dirPath);
155 
156  // print all the files and directories within directory
157  while ((ent = readdir(dir)) != NULL) {
158  if(isFile(std::string(dirPath + "/" + ent->d_name))){
159  // Only add files with the requested file extension
160  if(ext != ""){
161  // Note: This can easily be fooled
162  std::string f_name(ent->d_name);
163  if(StrUtils::endsWithString(f_name, ext))
164  files.push_back(f_name);
165  }
166  // Don't care what the file extension is
167  else
168  files.emplace_back(ent->d_name);
169  }
170  }
171  closedir(dir);
172 
173  return files;
174 }

322 bool FSUtils::deleteDir(const std::string &dirpath){
323  if(!dexists(dirpath))
324  return true;
325 
326  DIR *dir;
327  struct dirent *ent;
328 
329  if((dir = opendir(dirpath.c_str())) == NULL)
330  throw std::runtime_error("Couldn't open " + dirpath + " for deletion");
331 
332  while((ent = readdir(dir)) != NULL){
333  std::string it(ent->d_name);
334 
335  // Be very careful not to traverse vertically out of the directory
336  if(THIS_DIR_DOT == it || PREV_DIR_DOT == it)
337  continue;
338 
339  std::string relative_it(dirpath + "/" + it);
340 
341  if(isDir(relative_it)){
342  // This really shouldn't ever be the case, since we couldn't be in this loop
343  if(dempty(relative_it))
344  std::remove(relative_it.c_str());
345  // Make the recursive call
346  else
347  deleteDir(relative_it);
348  }
349  // Remove all the files from the directory
350  else if(isFile(relative_it))
351  std::remove(relative_it.c_str());
352  else
353  throw std::runtime_error("Not sure what else we could have here...");
354  }
355  closedir(dir);
356 
357  if(isDir(dirpath) && dempty(dirpath))
358  std::remove(dirpath.c_str());
359 
360  return !dexists(dirpath);
361 }
