182  std::string timeStamp(){
183  auto now = std::chrono::system_clock::now();
184  auto count = std::chrono::system_clock::to_time_t(now);
185  auto secs = std::chrono::time_point_cast<std::chrono::seconds>(now);
186  auto partial = now - secs;
187  auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(partial);
188

195  void pop_and_write(){
196  // Main loop for the logger thread, will check queue for messages and write them
197  while(true){
198 
199 
200  // Use try_and_pop instead of wait_and_pop so the thread can be immediately killed
201  // if necessary; wait_and_pop will block on the thread while the queue is empty
202  logmessage_t msg;
203  bool did_receive_message = msg_queue_.try_and_pop(msg, timeout_);
204 
205  // Process the received message and write it to the log file
206  if(did_receive_message){
207  std::string time_str = timeStamp();
208  std::string fnamestr = "";
209  if(msg.function_name_ != "")
210  fnamestr = msg.function_name_;
211  if(fnamestr != "")
212  fnamestr += ": ";
213 
214  std::stringstream ss;
215  ss << time_str << " " << msg.log_message_type_ << ": " << fnamestr;
216  ss << msg.message_to_be_logged_ << '\n';
217  std::ofstream out(logFile_, std::ios::out | std::ios::app);
218  out << ss.str() << std::flush;
219  }
220 
221  // stop_logging_ set to true in d'tor, keep pulling messages until queue is empty
222  if(stop_logging_ && msg_queue_.empty())
223  break;
224 
225  // kill_ allows for immediate thread death regardless of messages already in queue
226  if(kill_)
227  break;
228  }
229  }
