95  void read_kernels_from_file(const char *fname, char **programBuffer){
97  /* FROM:
98  -> http://dhruba.name/2012/08/16/opencl-cookbook-creating-programs-and-reading-kernels-from-a-file/
99  */
100  // get size of kernel source
101  FILE *programHandle = fopen(fname, "r");
102  fseek(programHandle, 0, SEEK_END);
103  int programSize = (int)ftell(programHandle);
104  rewind(programHandle);
105 
106  // read kernel source into buffer
107  (*programBuffer) = (char*) malloc(programSize + 1);
108  (*programBuffer)[programSize] = '\0';
109  fread((*programBuffer), sizeof(char), programSize, programHandle);
110  fclose(programHandle);
111  }

146  void print_selected_device_info(cl_device_id cdi, int device_num){
147 
148  char* value;
149  size_t valueSize;
150  cl_uint maxComputeUnits;
151 
152  // print device name
153  clGetDeviceInfo(cdi, CL_DEVICE_NAME, 0, NULL, &valueSize);
154  value = (char*) malloc(valueSize);
155  clGetDeviceInfo(cdi, CL_DEVICE_NAME, valueSize, value, NULL);
156  printf("%d. Device: %s\n", device_num, value);
157  free(value);
158 
159  // print hardware device version
160  clGetDeviceInfo(cdi, CL_DEVICE_VERSION, 0, NULL, &valueSize);
161  value = (char*) malloc(valueSize);
162  clGetDeviceInfo(cdi, CL_DEVICE_VERSION, valueSize, value, NULL);
163  printf(" %d.%d Hardware version: %s\n", device_num, 1, value);
164  free(value);
165 
166  // print software driver version
167  clGetDeviceInfo(cdi, CL_DRIVER_VERSION, 0, NULL, &valueSize);
168  value = (char*) malloc(valueSize);
169  clGetDeviceInfo(cdi, CL_DRIVER_VERSION, valueSize, value, NULL);
170  printf(" %d.%d Software version: %s\n", device_num, 2, value);
171  free(value);
172 
173  // print c version supported by compiler for device
174  clGetDeviceInfo(cdi, CL_DEVICE_OPENCL_C_VERSION, 0, NULL, &valueSize);
175  value = (char*) malloc(valueSize);
176  clGetDeviceInfo(cdi, CL_DEVICE_OPENCL_C_VERSION, valueSize, value, NULL);
177  printf(" %d.%d OpenCL C version: %s\n", device_num, 3, value);
178  free(value);
179 
180  // print parallel compute units
181  clGetDeviceInfo(cdi, CL_DEVICE_MAX_COMPUTE_UNITS, sizeof(maxComputeUnits), &maxComputeUnits, NULL);
182  printf(" %d.%d Parallel compute units: %d\n", device_num, 4, maxComputeUnits);
183  }
